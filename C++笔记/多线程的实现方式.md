多线程的实现方式

demo1:

```C++
// 外部程序路径
QString program = "D:\\SimEngine\\build-SimEng-Desktop_Qt_5_12_7_MSVC2017_64bit-Debug\\debug\\JMASE_v1.0\\JMASE_DEV\\Teams\\Demo1\\bin\\Debug\\Demo1.exe";
// 命令行参数
QStringList args;
args << ""
     << "-scenario"
     << "D:\\SimEngine\\build-SimEng-Desktop_Qt_5_12_7_MSVC2017_64bit-Debug\\debug\\JMASE_v1.0\\JMASE_DEV\\Teams\\Demo1\\scenario0\\scenario0.jscn";
// 创建一个进程
QProcess *process = new QProcess;
// 运行外部程序
process->start(program, args);
// 等待程序结束，参数是最大等待时间（毫秒）
process->waitForFinished(10000);
// 读出外部程序cout或printf输出的字符
QByteArray output = process->readAllStandardOutput();
// 释放内存
delete process;

```
